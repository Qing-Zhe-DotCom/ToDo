Controller 模式（每个视图统一）

public class InfoPanelController {
  private final InfoPanelView view;
  private final ScheduleItemService service;
  
  // 职责：参数收集 → 调用 service → 更新 view
  // 禁止：直接写 SQL、长时间阻塞 UI
  public void showItem(String itemId) {
    service.findById(itemId).ifPresent(item -> {
      var vm = InfoPanelViewModel.from(item);
      view.bind(vm);
    });
  }
  public void dispose() { // 解绑、释放资源 }
}
